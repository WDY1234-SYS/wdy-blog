防抖：用户频繁的操作，但是我只要最后一次事件的操作

```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>防抖</title>
</head>

<body>
    <input type="text">
    <script>
        //防抖：用户频繁的操作，但是我只要最后一次事件的操作
        let inp = document.querySelector('input')
        let t = null
        inp.oninput = debounce(function () {
            //业务代码
            console.log(this.value);
        }, 500)

        //封装debounce方法
        function debounce(fn, delay) {
            return function () {
                if (t !== null) {
                    clearTimeout(t)
                }
                t = setTimeout(() => {
                    fn.call(this)
                }, delay)
            }
        }
    </script>
</body>

</html>
```

节流：控制执行次数

```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>节流</title>
</head>
<style>
    body {
        height: 2000px;
    }
</style>

<body>
    <script>
        //节流：控制执行次数
        window.onscroll = throttle(function () {
            //业务代码
            console.log('hello');
        }, 500)

        //封装throttle方法
        function throttle(fn, delay) {
            let flag = true;
            return function () {
                if (flag) {
                    setTimeout(() => {
                        fn.call(this)
                        flag = true
                    }, delay)
                }
                flag = false
            }
        }


    </script>
</body>

</html>
```
